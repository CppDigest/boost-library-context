Take the following code:  
  
```c++  
#include <boost/geometry/algorithms/correct.hpp>  
#include <boost/geometry/algorithms/difference.hpp>  
#include <boost/geometry/geometries/multi_polygon.hpp>  
#include <boost/geometry/geometries/point_xy.hpp>  
#include <boost/geometry/geometries/polygon.hpp>  
#include <boost/geometry/geometries/ring.hpp>  
  
int main()  
{  
  using BPoint = boost::geometry::model::d2::point_xy<float, boost::geometry::cs::cartesian>;  
  using BPoly = boost::geometry::model::polygon<BPoint>;  
  using BMulti = boost::geometry::model::multi_polygon<BPoly>;  
  
  // 1  
  boost::geometry::model::ring<BPoint> ring;  
  boost::geometry::correct(ring);  
  
  // 2  
  BMulti geom, out;  
  boost::geometry::difference(geom, ring, out);  
}  
```

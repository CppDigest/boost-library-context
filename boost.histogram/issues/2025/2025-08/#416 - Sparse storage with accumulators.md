# #416 - Sparse storage with accumulators [Open]

> Username: henryiii  
> Created at: 2025-08-15 20:09:55 UTC  
> Updated at: 2025-08-15 20:10:49 UTC  
> Url: https://github.com/boostorg/histogram/issues/416  

Currently, sparse storage works:  
  
```cpp  
#include <boost/histogram.hpp>  
#include <boost/histogram/ostream.hpp>  
#include <unordered_map>  
#include <iostream>  
#include <vector>  
  
int main() {  
    namespace bh = boost::histogram;  
    auto h = bh::make_histogram_with(std::unordered_map<std::size_t, double>(), bh::axis::regular<>(10, 0.0, 1.0));  
    std::vector<double> values = {.2, .3};  
    std::vector<double> weights = {1, 1};  
    h.fill(values, bh::weight(weights));  
    std::cout << bh::algorithm::sum(h) << std::endl;  
}  
```  
  
As does accumulators with dense storage:  
  
```cpp  
#include <boost/histogram.hpp>  
#include <boost/histogram/ostream.hpp>  
#include <iostream>  
#include <vector>  
  
int main() {  
    namespace bh = boost::histogram;  
    auto h = bh::make_histogram_with(std::vector<bh::accumulators::weighted_sum<double>>(), bh::axis::regular<>(10, 0.0, 1.0));  
    std::vector<double> values = {.2, .3};  
    std::vector<double> weights = {1, 1};  
    h.fill(values, bh::weight(weights));  
    std::cout << bh::algorithm::sum(h) << std::endl;  
}  
```  
  
  
But you can't use the two together:  
  
```cpp  
#include <boost/histogram.hpp>  
#include <boost/histogram/ostream.hpp>  
#include <unordered_map>  
#include <iostream>  
#include <vector>  
  
int main() {  
    namespace bh = boost::histogram;  
    auto h = bh::make_histogram_with(std::unordered_map<std::size_t, bh::accumulators::weighted_sum<double>>(), bh::axis::regular<>(10, 0.0, 1.0));  
    std::vector<double> values = {.2, .3};  
    std::vector<double> weights = {1, 1};  
    h.fill(values, bh::weight(weights));  
    std::cout << bh::algorithm::sum(h) << std::endl;  
}  
```

### What's going wrong + Context  
I can't give out much, but basically the websocket portion of this code is externally controlled by another program which uses a separate mechanism slipped into the io_context to have it stop, start, and write messages out across the websocket.  That is to say, the synchronous write call is actually being called from within the io_context from another callback that was placed in via boost::post.  As such, everything in this code *should* be async by nature.  Same for close_connection.  Furthermore, the callback that is utilizing write can only execute write if socket_open is returning true.    
  
The piece I seem to find conflicting information on is what one is supposed to do to properly shut down the websocket.  I am more familiar with the TCP side of boost asio and have the basic ideas down (or so I think)...when the socket needs closed, you cancel all pending operations, send shutdown, and then read until you hit an error, at which point you close the socket.  But for websocket, the best I could gather was that cancel on the lower layer was ill advised since under some conditions the websocket will close itself and you don't wish to cancel that operation.  So always use async_close was what I thought the wisdom was.  But the more problems I fix the more I struggle to understand the best practice when it comes to shutting down websocket specifically, whether it's from running into an error whilst reading or closing it by force.  
  
I recognize a couple things that I will fix, such as when I run into the write failure, I should kick off an async_close, but I find it interesting the crash appears to be in the do_read function as opposed to the write call, and both times I've seen this error it always happens after the write call but the crash is in do_read.  Any and all insight on this would be much appreciated.  Thanks!

            });  
}  
  
void Websocket::on_tcp_accept(boost::system::error_code ec) {  
    VLOG(OCCASIONAL_ASYNC) << "TCP acceptor callback triggered with: " << ec.message();  
    if (ec == boost::asio::error::operation_aborted) { return; }  
    if (ec) {   
        LOG(WARNING) << "Websocket connection failed: " << ec.message();   
        _acceptor.close();  
        do_tcp_accept();  
        return;  
    }  
    VLOG(OCCASIONAL_ASYNC) << "Websocket connection accepted: " << _tcpSocket.remote_endpoint();

    _websocket->async_accept(

            auto selfShared = selfWeak.lock();  
            if (not selfShared) { return; }  
            selfShared->on_websocket_accept(ec);  
        });  
    _acceptor.close();  
}  
  
void Websocket::on_websocket_accept(boost::system::error_code ec) {  
    if (ec) {  
        LOG(WARNING) << "Websocket handshake failed: " << ec.message();  
        return;  
    }  
    VLOG(OCCASIONAL_ASYNC) << "Websocket handshake accepted: " << _websocket->next_layer().remote_endpoint();  
    VLOG(OCCASIONAL_ASYNC) << "Listening for messages...";  
    do_read();  
}  
  
void Websocket::do_read() {  
    _buffer.consume(_buffer.size());  
    _websocket->async_read(_buffer,   
        [selfWeak=std::weak_ptr<Websocket>(shared_from_this())](boost::system::error_code ec, size_t bytes_read) {  
            auto selfShared = selfWeak.lock();  
            if (not selfShared) { return; }  
            selfShared->on_read(ec, bytes_read);  
        });  
}  
  
void Websocket::on_read(boost::system::error_code ec, size_t bytes_read) {  
    VLOG(FREQ_ASYNC) << "Message received with " << bytes_read << " bytes read.";  
    if(early_return_read_errors(ec, bytes_read)) { return; }  
    // We do some stuff here with the message....  
    do_read();  
}  
  
bool Websocket::early_return_read_errors(boost::system::error_code ec, const size_t& bytes_read) {  
    if (ec == boost::asio::error::operation_aborted || _closing ) { return true; }  
    if (ec == boost::beast::websocket::error::closed) {  
        VLOG(OCCASIONAL_ASYNC) << "Websocket closed error code received, closing underlying connection";  
        close_underlying();  
        return true;  
    }  
    if (ec == boost::asio::error::eof ||   
        ec == boost::beast::websocket::condition::protocol_violation ||  
        ec == boost::asio::error::broken_pipe) {  
        LOG(WARNING) << ec.message() << " encountered, closing websocket connection.";  
        close_connection();  
        return true;  
    }  
    if (ec) {  
        LOG(WARNING) << "Websocket read failed: " << ec.message() << " ignoring message.";  
        do_read();  
        return true;  
    }  
    if (bytes_read == 0) {  
        LOG(WARNING) << "0 bytes read, ignoring message";  
        do_read();  
        return true;  
    }  
    if (not _websocket->got_binary()) {  
        LOG(WARNING) << "Unexpected Websocket 'text' packet, skipping message...";  
        do_read();  
        return true;  
    }  
    return false;  
}  
  
bool Websocket::socket_open() {   
    return (_websocket && _websocket->is_open() && _websocket->next_layer().is_open() && not _closing);   
}  
  
void Websocket::write(const unsigned char* data, size_t size) try {  
    boost::system::error_code ec;  
    _websocket->binary(true);  
    auto numBytesWritten = _websocket->write(boost::asio::buffer(data, size), ec);  
    if (ec) { LOG(WARNING) << "Write error code: " << ec.message(); }  
    if (size != numBytesWritten) {   
        LOG(WARNING) << "Write call bytes written not equal to passed in size, "   
                    << size << " expected vs " << numBytesWritten << " written.";  
    }  
}  
catch (std::exception& e) {  
    LOG(WARNING) << "Websocket write failed, DDS message dropped: " << e.what();  
}  
  
void Websocket::close_connection() {  
    if (_closing) { return; }  
    VLOG(OCCASIONAL_ASYNC) << "Deactivating websocket server...";  
    // In case we never connected...  
    boost::system::error_code ec;  
    if (_acceptor.is_open()) {   
        VLOG(OCCASIONAL_ASYNC) << "Closing acceptor...";  
        _acceptor.close(ec);   
        VLOG(OCCASIONAL_ASYNC) << "Acceptor close error code: " << ec.message();  
        return;  
    }  
    do_close();  
}  
  
void Websocket::do_close() {  
    if (not websocket_closable()) { return; }  
    _closing = true;  
    _websocket->async_close(boost::beast::websocket::close_code::none,   
        [selfWeak=std::weak_ptr<Websocket>(shared_from_this())](boost::system::error_code ec) {  
            auto selfShared = selfWeak.lock();  
            if (not selfShared) { return; }  
            LOG(INFO) << "Async close ec: " << ec.message();  
            selfShared->close_underlying();  
        });  
}  
  
void Websocket::close_underlying() {  
    boost::system::error_code ec;  
    if (_websocket && _websocket->next_layer().is_open()) {  
        _websocket->next_layer().shutdown(boost::asio::ip::tcp::socket::shutdown_both, ec);  
        LOG(INFO) << "TCP socket shutdown error message return: " << ec.message();  
        _websocket->next_layer().close(ec);  
        LOG(INFO) << "Websocket next layer closure error message return: " << ec.message();  
    }  
    if (_websocket && _websocket->lowest_layer().is_open()) {  
        _websocket->lowest_layer().close(ec);  
        LOG(INFO) << "Websocket lowest layer closure error message return: " << ec.message();  
    }  
    _closing = false;  
}  
  
bool Websocket::websocket_closable() {  
    bool socketClosable = true;  
    VLOG(OCCASIONAL_ASYNC) << "Checking for old websocket...";  
    if (not _websocket) { socketClosable = false; }  
    else { if (not _websocket->is_open()) { socketClosable = false; }}  
    set_close_status();  
    _closing = false;  
    return socketClosable;  
}

void Websocket::do_tcp_accept() {  
    if (_sessionActive) {  
        std::cout << "Websocket session already exists, ignoring request to take another connection." << std::endl;  
        return;  
    }  
    if (_listening) {  
        std::cout << "Websocket session not active, but already listening for a connection, ignoring request to listen again." << std::endl;  
        return;  
    }  
    boost::system::error_code ec;  
    if (not _acceptor.is_open()) {  
        _acceptor.open(boost::asio::ip::tcp::v4(), ec);  
        if (ec) { std::cout << "Failed to open tcp acceptor: " << ec.message() << std::endl; }  
        _acceptor.set_option(boost::asio::socket_base::reuse_address(true), ec);  
        if (ec) { std::cout << "Failed to set tcp acceptor reuse address option: " << ec.message() << std::endl; }  
        _acceptor.bind(boost::asio::ip::tcp::endpoint(boost::asio::ip::tcp::v4(), _port), ec);  
        if (ec) { std::cout << "Failed to bind tcp acceptor: " << ec.message() << std::endl; }  
        _acceptor.listen(0, ec);  
        if (ec) { std::cout << "Failed to listen on tcp acceptor: " << ec.message() << std::endl; }  
        _listening = true;  
    }  
    std::cout << "Websocket Server listening for connections." << std::endl;  
    _acceptor.async_accept(  
            _tcpSocket,   
            [selfWeak=std::weak_ptr<Websocket>(shared_from_this())](boost::system::error_code ec) {  
                auto selfShared = selfWeak.lock();  
                if (not selfShared) { return; }  
                selfShared->on_tcp_accept(ec);  
            });  
}  
  
void Websocket::on_tcp_accept(boost::system::error_code ec) {  
    _listening = false;  
    std::cout << "TCP acceptor callback triggered with: " << ec.message() << std::endl;  
    if (ec == boost::asio::error::operation_aborted) { return; }  
    if (ec) {   
        std::cout << "Websocket connection failed: " << ec.message() << std::endl;   
        _acceptor.close();  
        do_tcp_accept();  
        return;  
    }  
    std::cout << "Websocket connection accepted: " << _tcpSocket.remote_endpoint() << std::endl;  
    _websocket = std::make_shared<boost::beast::websocket::stream<boost::asio::ip::tcp::socket>>(std::move(_tcpSocket));  
    _sessionActive = true;  
    // If we update our boost libraries to 1.70, add this in and get rid of the next_layer() cancel call in close, may  
    // be the proper procedure...  
    // _websocket->set_option(boost::beast::websocket::stream_base::timeout::suggested(boost::beast::role_type::server));  
    _websocket->async_accept(  
        [selfWeak=std::weak_ptr<Websocket>(shared_from_this())](boost::system::error_code ec) {  
            auto selfShared = selfWeak.lock();  
            if (not selfShared) { return; }  
            selfShared->on_websocket_accept(ec);  
        });  
    // Don't want anymore clients once we have accepted one without error  
    _acceptor.close();  
}  
  
void Websocket::on_websocket_accept(boost::system::error_code ec) {  
    if (ec) {  
        std::cout << "Websocket handshake failed: " << ec.message() << std::endl;  
        _sessionActive = false;  
        return;  
    }  
    std::cout << "Websocket handshake accepted: " << _websocket->next_layer().remote_endpoint() << std::endl;  
    std::cout << "Listening for messages." << std::endl;  
    do_read();  
}  
  
void Websocket::do_read() {  
    _buffer.consume(_buffer.size());  
    _websocket->async_read(_buffer,   
        [selfWeak=std::weak_ptr<Websocket>(shared_from_this())](boost::system::error_code ec, size_t bytes_read) {  
            auto selfShared = selfWeak.lock();  
            if (not selfShared) { return; }  
            selfShared->on_read(ec, bytes_read);  
        });  
}  
  
void Websocket::on_read(boost::system::error_code ec, size_t bytes_read) {  
    std::cout << "Message received with " << bytes_read << " bytes read." << std::endl;  
  
    if(early_return_read_errors(ec, bytes_read)) { return; }  
    std::cout << "Sending message: stuff" << std::endl;  
    buffer_to_vector();  
    // Stuff sending message via non boost method  
    do_read();  
}  
  
bool Websocket::early_return_read_errors(boost::system::error_code ec, const size_t& bytes_read) {  
    if (ec) {  
        std::cout << "Websocket read error code: " << ec.message() << std::endl;  
        close_connection();  
        return true;  
    }  
    if (_closing) {  
        std::cout << "Websocket closure in progress..." << std::endl;  
        return true;  
    }  
    // TODO: Uncertain if we should retry read or just close connection for the   
    // following scenarios  
    if (bytes_read == 0) {  
        std::cout << "0 bytes read, closing connection." << std::endl;  
        close_connection();  
        return true;  
    }  
    // Never expecting this to happen, but just in case...  
    if (not _websocket->got_binary()) {  
        std::cout << "Unexpected Websocket 'text' packet, skipping message..." << std::endl;  
        do_read();  
        return true;  
    }  
    return false;  
}

}

    boost::system::error_code ec;

    }

}

    }  
    do_close();  
}

        });

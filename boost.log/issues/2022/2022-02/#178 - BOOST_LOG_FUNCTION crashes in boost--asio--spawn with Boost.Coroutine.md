void init_logging() {  
  auto core = boost::log::core::get();  
  auto backend = boost::make_shared<boost::log::sinks::text_ostream_backend>();  
  backend->add_stream(  
      boost::shared_ptr<std::ostream>(&std::cout, boost::null_deleter{}));  
  backend->auto_flush(true);  
  auto sink = boost::make_shared<boost::log::sinks::synchronous_sink<  
      boost::log::sinks::text_ostream_backend>>(backend);  
  core->add_sink(sink);  
  
  namespace expr = boost::log::expressions;  
  namespace attrs = boost::log::attributes;  
  namespace keywords = boost::log::keywords;  
  
  sink->set_formatter(expr::stream  
                      << "["  
                      << expr::format_named_scope(  
                             "Scope", keywords::format = "%F:%l: %C",  
                             keywords::iteration = expr::reverse)  
                      << "]: " << expr::smessage);  
  
  core->add_global_attribute("Scope", attrs::named_scope{});  
}  
  
void hanlde_async_client(  
    std::shared_ptr<boost::asio::ip::tcp::socket> tcp_stream,  
    boost::asio::yield_context yield) {  
  BOOST_LOG_FUNCTION();  
  std::string buf;  
  buf.resize(80);  
  const std::size_t tcp_read_len =  
      tcp_stream->async_receive(boost::asio::buffer(buf), yield);  
  buf.resize(tcp_read_len);  
  BOOST_LOG_TRIVIAL(debug) << "Read: " << buf;  
}  
  
void async_sever(boost::asio::io_context &io_context,  
                 const boost::asio::ip::port_type port,  
                 boost::asio::yield_context yield) {  
  BOOST_LOG_FUNCTION();  
  boost::asio::ip::tcp::acceptor tcp_listener{io_context};  
  const boost::asio::ip::tcp::endpoint tcp_listen_addr{  
      boost::asio::ip::make_address("127.0.0.1"), port};  
  tcp_listener.open(tcp_listen_addr.protocol());  
  tcp_listener.set_option(boost::asio::ip::tcp::acceptor::reuse_address(true));  
  tcp_listener.bind(tcp_listen_addr);  
  tcp_listener.listen();  
  BOOST_LOG_TRIVIAL(info) << "Listening on " << tcp_listener.local_endpoint();  
  
  for (int i = 0; i < 1; i++) {  
    boost::asio::ip::tcp::socket tcp_stream{io_context};  
    tcp_listener.async_accept(tcp_stream, yield);  
    auto tcp_stream_ptr =  
        std::make_shared<boost::asio::ip::tcp::socket>(std::move(tcp_stream));  
    BOOST_LOG_TRIVIAL(info)  
        << "Incoming connection from " << tcp_stream_ptr->remote_endpoint();  
    boost::asio::spawn(io_context,  
                       [tcp_stream_ptr](boost::asio::yield_context yield) {  
                         hanlde_async_client(tcp_stream_ptr, yield);  
                       });  
  }  
}  
  
int main() {

                     [&io_context](boost::asio::yield_context yield) {  
                       async_sever(io_context, 5000, yield);  
                     });  
  io_context.run();  
}

I'm not very familiar with coroutines and Boost.Coroutine in particular, but from what I understand `boost::asio::spawn` creates a new stack for the coroutine to run. This means thread-specific attributes that reference the values on stack (such as named scopes) cannot work with stackful coroutines, as as soon as the stack is replaced the references are dangling.  
  
There is no built-in attribute similar to named scopes that doesn't rely on the stack, mainly for efficiency reasons. However, you can implement [an attribute of your own](https://www.boost.org/doc/libs/1_78_0/libs/log/doc/html/log/extension/attributes.html).  
  
[C++20 coroutines](https://en.cppreference.com/w/cpp/language/coroutines) are stackless, so if I understand it correctly, they should be compatible with named scopes.

## Comment 4

> Username: Lastique  
> Created at: 2022-02-10 20:08:16 UTC  
> Updated at: 2022-02-10 21:02:32 UTC  
> Url: https://github.com/boostorg/log/issues/178#issuecomment-1035443941  

The problem with named scopes is that the implementation assumes LIFO behavior of named scopes list. That is, when a named scope sentry is destroyed, the most recently pushed entry is removed. This is expected in normal operation - you leave scopes in the opposite order than you had entered. This is not the case with coroutines - you may enter one scope, switch context to another coroutine, enter another scope in that context, then switch back and leave the first scope. The implementation will attempt to pop the second scope instead, which will leave the list in inconsistent state as it will still reference the (now destroyed) first scope entry.  
  
Then there is the part where the coroutine stack is freed - I'm not sure at what point this happens and what happens with the objects that are stored on the stack. I assume, their destructors should be called, but with the LIFO requirement described above, it is important when the destructors are called and in what order.  
  
All in all, named scopes just don't make sense with coroutines, stackless or not, and should probably not be used anyway. This mechanism simply does not provide any reliable information wrt. execution backtrace with coroutines.
